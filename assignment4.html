def ndb_search(search,fresh = False,asdf = False):
    """This function takes in a search term (1 string) and passes it directly to the
    USDA Food Composition Database. It returns a list of items, or an empty list if the search
    term fails to return valid items."""
    api = "4y8emwdeDUa2nZkAdQRMN5sBhTu6g8Y1HtFL0fkM"
    url = "https://api.nal.usda.gov/ndb/search/?format=json&q=%s&sort=r&max=25&offset=0&api_key=%s" 
    queryurl = url % (search,api)
    if fresh == True:
        ds = "&ds=Standard Reference"
        url = "https://api.nal.usda.gov/ndb/search/?format=json&q=%s%s&sort=r&max=25&offset=0&api_key=%s"
        queryurl = url % (search,ds,api)
    #This formats the url for the api
    requests_cache.install_cache('food_cache')
    r = requests.get(queryurl)
    #get request and creates a cache for requests to speed up repeat requests
    data = pandas.read_json(r.text)
    data = data.T
    #Creates a dataframe (makes it easier to format and extract data)
    if len(data.iloc[0]) == 1:
        return list()
        #Error checking for invalid search terms
    df = data['item']
    #gets list
    if asdf == True:
        df = pandas.DataFrame(df[0])
        return df
    return df[0]

def findfoods(food):
    """Takes String Gives DataFrame"""
    append = " raw"
    search = food + append
    item = ndb_search(search,True,True)
    if len(item)==0:
        x = pandas.DataFrame( "none",index = [random.randint(100,10000000)], columns= ["ds","group","name","ndbno","offset"])
        return x
    else:
        return item.iloc[:1]

def ndb_report(search,asdf = False):
    """This function takes in a search term (1 string) and passes it directly to the
    USDA Food Composition Database. """
    api = "4y8emwdeDUa2nZkAdQRMN5sBhTu6g8Y1HtFL0fkM"
    url = "https://api.nal.usda.gov/ndb/V2/reports?ndbno=%s&type=b&format=json&api_key=%s" 
    queryurl = url % (search,api)
    #This formats the url for the api
    requests_cache.install_cache('food_cache2')
    r = requests.get(queryurl)
    #get request and creates a cache for requests to speed up repeat requests
    data = pandas.read_json(r.text)
    #Creates a dataframe (makes it easier to format and extract data)
    df = data['foods'][0]
    #gets list
    df = pandas.DataFrame(df).T
    if asdf == True:
        return pandas.DataFrame(df['nutrients'][0])
    return df['nutrients'][0]
