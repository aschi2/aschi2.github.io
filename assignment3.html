def datareader(target):
    data = pd.DataFrame()
    #Create a pandas df
    for f in glob.glob(("../Homework 3/assignment3_data/%s/*.xls*" % target)):
        #glob lets us grab the names of all files with a certain format. 
        #This allows the function to able to look for all .xls or .xlsx files
        #in the data folder by just specifying the file name.
        df = pd.read_excel(f,skiprows=range(3),header = None)
        #skips header info
        df = df.drop(df.columns[[2,5]],axis = 1)
        #drops columns for formatting
        df = df.dropna()
        #drop rows that have NA, this gets rid of padder rows at the
        #end of the file
        item = re.search('\\\(.+)\.xlsx?',f).group(1)
        #uses regex to find the name of the file that we are currently in
        df['2'] = item
        df['7'] = target
        data = data.append(df,ignore_index=True)
        #append the entry with food and type
    data = data.drop(data.columns[[5,6]],axis = 1)
    #drop columns for formatting
    data = data.dropna(axis=0)
    #drop NA's for formatting
    data.columns = ['form','price_pre_lb','yield','lb_per_cup','price_per_cup','food','type']
    #label columns
    data = data[data.form.str.contains('fresh|Fresh')]
    #search for fresh in the form column in order to find only fresh versions of produce
    data = data.reset_index(drop=True)
    #reset the indexes that are misplaced due to all the splicing and dropping
    #of rows.
    return data
    
data = datareader("fruit")
